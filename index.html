function showSection(section) {

// Hide all sections

document.querySelectorAll('.content-section').forEach(el => {

el.classList.remove('active');

});

// Remove active class from buttons

document.querySelectorAll('.nav-btn').forEach(el => {

el.classList.remove('active');

});

// Show selected section

document.getElementById(section).classList.add('active');

event.target.classList.add('active');

}

function renderWorkouts() {

Object.keys(workouts).forEach(day => {

const container = document.getElementById(day + '-exercises');

if (!container) return;

workouts[day].forEach((exercise, index) => {

const card = createExerciseCard(exercise, index + 1);

container.appendChild(card);

});

});

}

function createExerciseCard(exercise, number) {

const div = document.createElement('div');

div.className = 'exercise-card';

const setBadges = exercise.sets.map((set, i) =>

const videoLink = exercise.video ?

`<a href="${exercise.video}" target="_blank" class="exercise-video" title="Watch Video Tutorial"></a>` : '';

div.innerHTML = `

</div>

<div class="exercise-stats">

<div class="stat">

function renderMeals() {

const container = document.getElementById('meals-container');

Object.keys(meals).forEach(key => {

const category = meals[key];

const section = document.createElement('div');

section.className = 'meal-category';

const nameMap = {

breakfast: 'Breakfast (475 kcal)',

preworkout: 'Pre-Workout (425 kcal)',

postworkout: 'Post-Workout (575 kcal)',

lunch: 'Lunch & Dinner (555-590 kcal)',

nonTraining: 'Meal 5 - Non Training (495 kcal)'

};

section.innerHTML = `

<div class="meal-header">

<span style="font-size: 1.5rem;">${category.icon}</span>

<span class="meal-type">${nameMap[key]}</span>

</div>

<div class="meals-grid" id="meals-${key}"></div>

`;

container.appendChild(section);

const grid = section.querySelector('.meals-grid');

category.options.forEach((meal, idx) => {

const card = createMealCard(meal, key, idx);

grid.appendChild(card);

});

});

}

function createMealCard(meal, category, index) {

const div = document.createElement('div');

div.className = 'meal-card';

div.onclick = () => toggleMeal(div);

const ingredientsList = Object.entries(meal.ingredients)

.map(([name, amount]) =>

`<div class="ingredient">

<span class="ingredient-name">${name}</span>

<span class="ingredient-amount">${amount}</span>

</div>`

).join('');

const instructions = meal.instructions ?

`<div style="margin-top: 0.75rem; padding-top: 0.75rem; border-top: 1px solid rgba(255,255,255,0.1); font-size: 0.8rem; color: var(--warning);">

<strong>Note:</strong> ${meal.instructions}

</div>` : '';

div.innerHTML = `

<div class="meal-name">${meal.name}</div>

<div class="meal-ingredients">

${ingredientsList}

${instructions}

</div>

`;

return div;

}

function toggleMeal(element) {

element.classList.toggle('selected');

}
